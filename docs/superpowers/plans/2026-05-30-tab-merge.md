// ── Tab 1：分條/切勾 ──────────────────────────────────────────────────────────
const _tab1 = { slitting: null, cutting: null };

function loadTab1() {
  jsonpCall(GAS_URL + '?action=getSlittingReport', d => { _tab1.slitting = d; _tryRenderTab1(); });
  jsonpCall(GAS_URL + '?action=getCuttingReport',  d => { _tab1.cutting  = d; _tryRenderTab1(); });
}

function _tryRenderTab1() {
  if (!_tab1.slitting || !_tab1.cutting) return;
  renderTab1(_tab1.slitting, _tab1.cutting);
}

function renderTab1(slit, cut) {
  const { monthly: sm, daily: sd, machineCounts = {}, currentMonth, currentYear } = slit;
  const { monthly: cm, daily: cd } = cut;
  const cyear  = currentYear;
  const cmonth = currentMonth;

  document.getElementById('sc-year-label').textContent  = `${cyear} 年分條／切勾月報`;
  document.getElementById('sc-daily-title').textContent = `▼ ${cyear}年 ${cmonth}月 日報`;

  // cutting monthly map: month number → row
  const cutMonthMap = {};
  cm.forEach(m => { cutMonthMap[Number(m.month.split('-')[1])] = m; });

  // 月報 rows
  let ytdR = 0, ytdE = 0, ytdH = 0, ytdCQ = 0, ytdCH = 0, ytdOK = false;
  const mrows = sm.map(m => {
    const c = cutMonthMap[m.month] || null;
    if (!m.hasData && !c) {
      return `<tr class="no-data text-center"><td class="text-dark fw-semibold">${m.month} 月</td>
        <td class="col-inv">—</td><td class="col-inv">—</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td></tr>`;
    }
    const jPerP = m.hasData && m.weightedHours > 0 ? m.rolls / m.weightedHours : null;
    const oPerP = m.hasData && m.weightedHours > 0 ? m.effRolls / m.weightedHours : null;
    const eff   = oPerP !== null ? oPerP / TARGET_SLIT : null;
    const cls   = eff === null ? '' : eff >= 1 ? 'eff-high' : eff >= 0.9 ? 'eff-mid' : 'eff-low';
    if (m.hasData) { ytdR += m.rolls; ytdE += m.effRolls; ytdH += m.weightedHours; ytdOK = true; }
    if (c) { ytdCQ += c.totalCount; ytdCH += c.totalWorkH; }
    return `<tr class="text-center">
      <td class="fw-semibold">${m.month} 月</td>
      <td class="text-end col-inv">${m.hasData ? fmt0(m.rolls) : '—'}</td>
      <td class="text-end col-inv">${jPerP !== null ? jPerP.toFixed(2) : '—'}</td>
      <td class="text-end">${m.hasData ? fmt2(m.effRolls) : '—'}</td>
      <td class="text-end">${m.hasData ? fmt2(m.weightedHours) : '—'}</td>
      <td class="text-end">${oPerP !== null ? oPerP.toFixed(2) : '—'}</td>
      <td class="${cls}">${eff !== null ? (eff * 100).toFixed(1) + '%' : '—'}</td>
      <td class="text-end">${c ? c.totalCount.toLocaleString() : '—'}</td>
      <td class="text-end">${c ? fmt2(c.totalWorkH) : '—'}</td>
      <td class="text-end">${c ? c.totalMachines : '—'}</td>
    </tr>`;
  });
  document.getElementById('sc-monthly-body').innerHTML = mrows.join('');

  // 月報 tfoot
  if (ytdOK && ytdH > 0) {
    const jPerP = ytdR / ytdH, oPerP = ytdE / ytdH;
    const eff = oPerP / TARGET_SLIT;
    const cls = eff >= 1 ? 'eff-high' : eff >= 0.9 ? 'eff-mid' : 'eff-low';
    document.getElementById('sc-monthly-foot').innerHTML = `<tr class="text-center">
      <td>年度累計</td>
      <td class="text-end col-inv">${fmt0(ytdR)}</td>
      <td class="text-end col-inv">${jPerP.toFixed(2)}</td>
      <td class="text-end">${fmt2(ytdE)}</td>
      <td class="text-end">${fmt2(ytdH)}</td>
      <td class="text-end">${oPerP.toFixed(2)}</td>
      <td class="${cls}">${(eff * 100).toFixed(1)}%</td>
      <td class="text-end">${ytdCQ.toLocaleString()}</td>
      <td class="text-end">${fmt2(ytdCH)}</td>
      <td class="text-end">—</td>
    </tr>`;
  }

  // 日報 rows
  const daysInMonth = new Date(cyear, cmonth, 0).getDate();
  const slitDayMap = {};
  sd.forEach(d => { slitDayMap[d.day] = d; });
  const cutDayMap = {};
  cd.forEach(d => { cutDayMap[Number(d.date.split('-')[2])] = d; });

  const today = new Date();
  let dR = 0, dE = 0, dH = 0, dCQ = 0, dCH = 0, dOK = false;
  const drows = [];

  for (let day = 1; day <= daysInMonth; day++) {
    const sl = slitDayMap[day];
    const cu = cutDayMap[day];
    const dayKey  = `${String(cmonth).padStart(2,'0')}/${String(day).padStart(2,'0')}`;
    const machines = machineCounts[dayKey] || 0;
    const dateObj  = new Date(cyear, cmonth - 1, day);
    const isToday  = cyear === today.getFullYear() && cmonth === today.getMonth()+1 && day === today.getDate();
    const isPast   = !isToday && (cyear < today.getFullYear() ||
      (cyear === today.getFullYear() && cmonth < today.getMonth()+1) ||
      (cyear === today.getFullYear() && cmonth === today.getMonth()+1 && day < today.getDate()));
    const isHoliday = !sl && !cu && (dateObj.getDay() === 0 || isPast);

    if (sl || cu) {
      if (sl) { dR += sl.rolls; dE += sl.effRolls; dH += sl.weightedHours; dOK = true; }
      if (cu) { dCQ += cu.count; dCH += cu.workH; }
      const jPerP = sl && sl.weightedHours > 0 ? sl.rolls / sl.weightedHours : null;
      const oPerP = sl && sl.weightedHours > 0 ? sl.effRolls / sl.weightedHours : null;
      const eff   = oPerP !== null ? oPerP / TARGET_SLIT : null;
      const cls   = eff === null ? '' : eff >= 1 ? 'eff-high' : eff >= 0.9 ? 'eff-mid' : 'eff-low';
      drows.push(`<tr>
        <td class="text-center fw-semibold">${cmonth}/${day}</td>
        <td class="text-center">${machines || '—'}</td>
        <td class="text-end col-inv">${sl ? fmt0(sl.rolls) : '—'}</td>
        <td class="text-end col-inv">${jPerP !== null ? jPerP.toFixed(2) : '—'}</td>
        <td class="text-end">${sl ? fmt2(sl.effRolls) : '—'}</td>
        <td class="text-end">${sl ? fmt2(sl.weightedHours) : '—'}</td>
        <td class="text-end">${oPerP !== null ? oPerP.toFixed(2) : '—'}</td>
        <td class="text-center ${cls}">${eff !== null ? (eff*100).toFixed(1)+'%' : '—'}</td>
        <td class="text-end">${cu ? cu.count.toLocaleString() : '—'}</td>
        <td class="text-end">${cu ? fmt2(cu.workH) : '—'}</td>
        <td class="text-end">${cu ? cu.machines : '—'}</td>
      </tr>`);
    } else if (isHoliday) {
      drows.push(`<tr class="day-holiday text-center">
        <td class="fw-semibold">${cmonth}/${day}</td><td>—</td>
        <td class="col-inv">—</td><td class="col-inv">—</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td>
      </tr>`);
    } else {
      drows.push(`<tr class="no-data text-center">
        <td class="text-dark fw-semibold">${cmonth}/${day}</td><td>—</td>
        <td class="col-inv">—</td><td class="col-inv">—</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td>
      </tr>`);
    }
  }
  document.getElementById('sc-daily-body').innerHTML = drows.join('');

  // 日報 tfoot
  if (dOK && dH > 0) {
    const jPerP = dR / dH, oPerP = dE / dH;
    const eff = oPerP / TARGET_SLIT;
    const cls = eff >= 1 ? 'eff-high' : eff >= 0.9 ? 'eff-mid' : 'eff-low';
    document.getElementById('sc-daily-foot').innerHTML = `<tr class="text-center">
      <td>月合計</td><td>—</td>
      <td class="text-end col-inv">${fmt0(dR)}</td>
      <td class="text-end col-inv">${jPerP.toFixed(2)}</td>
      <td class="text-end">${fmt2(dE)}</td>
      <td class="text-end">${fmt2(dH)}</td>
      <td class="text-end">${oPerP.toFixed(2)}</td>
      <td class="${cls}">${(eff*100).toFixed(1)}%</td>
      <td class="text-end">${dCQ.toLocaleString()}</td>
      <td class="text-end">${fmt2(dCH)}</td>
      <td class="text-end">—</td>
    </tr>`;
  }
}

const _tab2 = { welding: null, backing: null, process: null };

function loadTab2() {
  jsonpCall(GAS_URL + '?action=getWeldingReport', d => { _tab2.welding = d; _tryRenderTab2(); });

function _tryRenderTab2() {

  renderTab2(_tab2.welding, _tab2.backing, _tab2.process);

function renderTab2(weld, back, proc) {
  const { monthly: wm, daily: wd, currentMonth, currentYear } = weld;
  const { monthly: bm, daily: bd } = back;
  const { monthly: pm, daily: pd } = proc;
  const cyear  = currentYear;
  const cmonth = currentMonth;

  document.getElementById('wbp-year-label').textContent  = `${cyear} 年焊接／褙膠／裁片月報`;
  document.getElementById('wbp-daily-title').textContent = `▼ ${cyear}年 ${cmonth}月 日報`;
  if (TARGET_WELD > 0) document.getElementById('wbp-weld-badge').textContent = `焊接目標 ${TARGET_WELD}`;

  const backMonthMap = {};
  bm.forEach(m => { backMonthMap[m.month] = m; });
  const procMonthMap = {};
  pm.forEach(m => { procMonthMap[Number(m.month.split('-')[1])] = m; });

  function weldPct(cPerP) {
    if (!TARGET_WELD) return { str: '—', cls: '' };
    const eff = cPerP / TARGET_WELD;
    return { str: `${(eff*100).toFixed(1)}%`, cls: eff >= 1 ? 'eff-high' : eff >= 0.9 ? 'eff-mid' : 'eff-low' };
  }

  // 月報 rows
  let ytdWC = 0, ytdWH = 0, ytdBN = 0, ytdBH = 0, ytdOK = false;
  const mrows = wm.map(m => {
    const b = backMonthMap[m.month];
    const p = procMonthMap[m.month];
    if (!m.hasData && !b?.hasData && !p) {
      return `<tr class="no-data text-center"><td class="text-dark fw-semibold">${m.month} 月</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td></tr>`;
    }
    const wCPerP = m.hasData && m.weightedHours > 0 ? m.effScore / m.weightedHours : null;
    const wPct   = wCPerP !== null ? weldPct(wCPerP) : { str: '—', cls: '' };
    const bNPerO = b?.hasData && b.weightedHours > 0 ? b.effRolls / b.weightedHours : null;
    const bEff   = bNPerO !== null ? bNPerO / TARGET_BACK : null;
    const bCls   = bEff === null ? '' : bEff >= 1 ? 'eff-high' : bEff >= 0.9 ? 'eff-mid' : 'eff-low';
    const pUtil  = p ? p.utilization : null;
    const pCls   = pUtil === null ? '' : pUtil >= 90 ? 'eff-high' : pUtil >= 80 ? 'eff-mid' : 'eff-low';
    if (m.hasData) { ytdWC += m.effScore; ytdWH += m.weightedHours; ytdOK = true; }
    if (b?.hasData) { ytdBN += b.effRolls; ytdBH += b.weightedHours; }
    return `<tr class="text-center">
      <td class="fw-semibold">${m.month} 月</td>
      <td class="text-end">${m.hasData ? fmt2(m.effScore) : '—'}</td>
      <td class="text-end">${m.hasData ? fmt2(m.weightedHours) : '—'}</td>
      <td class="text-end">${wCPerP !== null ? wCPerP.toFixed(2) : '—'}</td>
      <td class="${wPct.cls}">${wPct.str}</td>
      <td class="text-end">${b?.hasData ? fmt2(b.effRolls) : '—'}</td>
      <td class="text-end">${b?.hasData ? fmt2(b.weightedHours) : '—'}</td>
      <td class="text-end">${bNPerO !== null ? bNPerO.toFixed(2) : '—'}</td>
      <td class="${bCls}">${bEff !== null ? (bEff*100).toFixed(1)+'%' : '—'}</td>
      <td class="text-end">${p ? fmt2(p.totalEffHours) : '—'}</td>
      <td class="text-end">${p ? fmt2(p.totalCutHours) : '—'}</td>
      <td class="${pCls}">${pUtil !== null ? pUtil.toFixed(1)+'%' : '—'}</td>
    </tr>`;
  });
  document.getElementById('wbp-monthly-body').innerHTML = mrows.join('');

  // 月報 tfoot
  if (ytdOK && ytdWH > 0) {
    const wCPerP = ytdWC / ytdWH;
    const wPct   = weldPct(wCPerP);
    const bNPerO = ytdBH > 0 ? ytdBN / ytdBH : 0;
    const bEff   = bNPerO / TARGET_BACK;
    const bCls   = bEff >= 1 ? 'eff-high' : bEff >= 0.9 ? 'eff-mid' : 'eff-low';
    document.getElementById('wbp-monthly-foot').innerHTML = `<tr class="text-center">
      <td>年度累計</td>
      <td class="text-end">${fmt2(ytdWC)}</td>
      <td class="text-end">${fmt2(ytdWH)}</td>
      <td class="text-end">${wCPerP.toFixed(2)}</td>
      <td class="${wPct.cls}">${wPct.str}</td>
      <td class="text-end">${fmt2(ytdBN)}</td>
      <td class="text-end">${fmt2(ytdBH)}</td>
      <td class="text-end">${bNPerO.toFixed(2)}</td>
      <td class="${bCls}">${(bEff*100).toFixed(1)}%</td>
      <td>—</td><td>—</td><td>—</td>
    </tr>`;
  }

  // 日報 rows
  const daysInMonth = new Date(cyear, cmonth, 0).getDate();
  const wDayMap = {}, bDayMap = {}, pDayMap = {};
  wd.forEach(d => { wDayMap[d.day] = d; });
  bd.forEach(d => { bDayMap[d.day] = d; });
  pd.forEach(d => { pDayMap[Number(d.date.split('-')[2])] = d; });

  const today = new Date();
  let dWC = 0, dWH = 0, dBN = 0, dBH = 0, dWOK = false;
  const drows = [];

  for (let day = 1; day <= daysInMonth; day++) {
    const wD = wDayMap[day], bD = bDayMap[day], pD = pDayMap[day];
    const dateObj  = new Date(cyear, cmonth - 1, day);
    const isToday  = cyear === today.getFullYear() && cmonth === today.getMonth()+1 && day === today.getDate();
    const isPast   = !isToday && (cyear < today.getFullYear() ||
      (cyear === today.getFullYear() && cmonth < today.getMonth()+1) ||
      (cyear === today.getFullYear() && cmonth === today.getMonth()+1 && day < today.getDate()));
    const isHoliday = !wD && !bD && !pD && (dateObj.getDay() === 0 || isPast);

    if (wD || bD || pD) {
      if (wD) { dWC += wD.effScore; dWH += wD.weightedHours; dWOK = true; }
      if (bD) { dBN += bD.effRolls; dBH += bD.weightedHours; }
      const wCPerP = wD && wD.weightedHours > 0 ? wD.effScore / wD.weightedHours : null;
      const wPct   = wCPerP !== null ? weldPct(wCPerP) : { str: '—', cls: '' };
      const bNPerO = bD && bD.weightedHours > 0 ? bD.effRolls / bD.weightedHours : null;
      const bEff   = bNPerO !== null ? bNPerO / TARGET_BACK : null;
      const bCls   = bEff === null ? '' : bEff >= 1 ? 'eff-high' : bEff >= 0.9 ? 'eff-mid' : 'eff-low';
      const pUtil  = pD ? pD.utilization : null;
      const pCls   = pUtil === null ? '' : pUtil >= 90 ? 'eff-high' : pUtil >= 80 ? 'eff-mid' : 'eff-low';
      drows.push(`<tr>
        <td class="text-center fw-semibold">${cmonth}/${day}</td>
        <td class="text-end">${wD ? fmt2(wD.effScore) : '—'}</td>
        <td class="text-end">${wD ? fmt2(wD.weightedHours) : '—'}</td>
        <td class="text-end">${wCPerP !== null ? wCPerP.toFixed(2) : '—'}</td>
        <td class="text-center ${wPct.cls}">${wPct.str}</td>
        <td class="text-end">${bD ? fmt2(bD.effRolls) : '—'}</td>
        <td class="text-end">${bD ? fmt2(bD.weightedHours) : '—'}</td>
        <td class="text-end">${bNPerO !== null ? bNPerO.toFixed(2) : '—'}</td>
        <td class="text-center ${bCls}">${bEff !== null ? (bEff*100).toFixed(1)+'%' : '—'}</td>
        <td class="text-end">${pD ? fmt2(pD.totalEffHours) : '—'}</td>
        <td class="text-end">${pD ? fmt2(pD.totalCutHours) : '—'}</td>
        <td class="text-center ${pCls}">${pUtil !== null ? pUtil.toFixed(1)+'%' : '—'}</td>
      </tr>`);
    } else if (isHoliday) {
      drows.push(`<tr class="day-holiday text-center">
        <td class="fw-semibold">${cmonth}/${day}</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td>
      </tr>`);
    } else {
      drows.push(`<tr class="no-data text-center">
        <td class="text-dark fw-semibold">${cmonth}/${day}</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td><td>—</td>
        <td>—</td><td>—</td><td>—</td>
      </tr>`);
    }
  }
  document.getElementById('wbp-daily-body').innerHTML = drows.join('');

  // 日報 tfoot
  if (dWOK && dWH > 0) {
    const wCPerP = dWC / dWH;
    const wPct   = weldPct(wCPerP);
    const bNPerO = dBH > 0 ? dBN / dBH : 0;
    const bEff   = bNPerO / TARGET_BACK;
    const bCls   = bEff >= 1 ? 'eff-high' : bEff >= 0.9 ? 'eff-mid' : 'eff-low';
    document.getElementById('wbp-daily-foot').innerHTML = `<tr class="text-center">
      <td>月合計</td>
      <td class="text-end">${fmt2(dWC)}</td>
      <td class="text-end">${fmt2(dWH)}</td>
      <td class="text-end">${wCPerP.toFixed(2)}</td>
      <td class="${wPct.cls}">${wPct.str}</td>
      <td class="text-end">${fmt2(dBN)}</td>
      <td class="text-end">${fmt2(dBH)}</td>
      <td class="text-end">${bNPerO.toFixed(2)}</td>
      <td class="${bCls}">${(bEff*100).toFixed(1)}%</td>
      <td>—</td><td>—</td><td>—</td>
    </tr>`;
  }
}
